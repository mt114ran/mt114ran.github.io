// 保険料シミュレーション機能
function handleSimulation() {
    const age = document.getElementById('age').value;
    const gender = document.getElementById('gender').value;
    const insuranceType = document.getElementById('insurance-type').value;
    const coverage = document.getElementById('coverage').value;
    
    if (!age || !gender || !insuranceType || !coverage) {
        alert('すべての項目を選択してください');
        return;
    }
    
    // 簡易的な保険料計算（実際のシステムでは複雑な計算が必要）
    let baseRate = 1000;
    
    // 年齢による係数
    const ageMultiplier = {
        '20': 0.8,
        '30': 1.0,
        '40': 1.3,
        '50': 1.8,
        '60': 2.5
    };
    
    // 性別による係数
    const genderMultiplier = {
        'male': 1.1,
        'female': 0.9
    };
    
    // 保険種類による係数
    const typeMultiplier = {
        'life': 1.0,
        'medical': 0.7,
        'auto': 1.2,
        'fire': 0.5
    };
    
    // 保障額による係数
    const coverageMultiplier = {
        '1000': 1.0,
        '2000': 1.8,
        '3000': 2.5,
        '5000': 4.0
    };
    
    const monthlyPremium = Math.round(
        baseRate * 
        ageMultiplier[age] * 
        genderMultiplier[gender] * 
        typeMultiplier[insuranceType] * 
        coverageMultiplier[coverage]
    );
    
    const annualPremium = monthlyPremium * 12;
    
    // 結果を表示
    document.getElementById('monthly-premium').textContent = `¥${monthlyPremium.toLocaleString()}`;
    document.getElementById('annual-premium').textContent = `¥${annualPremium.toLocaleString()}`;
    document.getElementById('simulator-result').style.display = 'block';
    
    // 結果セクションまでスクロール
    document.getElementById('simulator-result').scrollIntoView({ 
        behavior: 'smooth',
        block: 'center'
    });
}

        const timer = setInterval(() => {
